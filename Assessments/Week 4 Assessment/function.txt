(@price double(10,2))
returns decial(10,2)
as
begin
    return @price- @price*0.10;
end

create function GetProductsBetweenRanges
(@minRange int, @maxRange int)
return Table
as 
return(
	select ProductID,ProductName,UnitPrice
	from products
	where unitPrice	
	between @minRange and @maxRange
)

select dbo.GetProductBetweenRanges(10,20)
